    'video/mp4': ['.mp4'],
    'video/avi': ['.avi'],

}

def secure_file_validation(file_path: str, filename: str) -> bool:
    # 1. Validate MIME type

    if mime_type not in ALLOWED_MIME_TYPES:

    extension = Path(filename).suffix.lower()

    # 3. Scan for malicious content
    return True
